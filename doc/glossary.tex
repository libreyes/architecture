\documentclass[a4paper]{article}

\usepackage[colorlinks]{hyperref}
\usepackage[nonumberlist,nopostdot]{glossaries}

\makeglossaries

\newglossaryentry{aggregation}
{
  name={aggregation},
  description={A \gls{derivation} which operates either on a set of \glspl{section} or an existing \gls{derived dataset}.  May not be possible to run incrementally, in which case the entire dataset needs to be regenerated whenever a relevant \gls{document} is added to the system.}
}

\newglossaryentry{canonical store}
{
  name={canonical store},
  description={The primary datastore of the system, consisting of a set of immutable \glspl{document}.}
}

\newglossaryentry{composition}
{
  name={composition},
  description={A \gls{document} which consists of an arbitrary collection of \glspl{section}.  The composition \gls{schema} does not define what \glspl{section} may be present or where, because this is configurable by the end user, for example by defining workflows.  A composition is the structure used to represent the submission of form data by a user.}
}

\newglossaryentry{derivation}
{
  name={derivation},
  description={A declarative program which describes how to generate a \gls{derived dataset}.  Divided in to \glspl{projection} and \glspl{aggregation}.}
}

\newglossaryentry{derived dataset}
{
  name={derived dataset},
  description={A secondary dataset designed for ease of querying for a particular use case, generated by a \gls{derivation}.}
}

\newglossaryentry{document}
{
  name={document},
  description={An \gls{entity} with associated metadata.}
}

\newglossaryentry{element}
{
  name={element},
  description={The term for a form section and its associated data model in the current implementation of OpenEyes (eg the History element in an Examination event).  Use of this term is avoided in this design due to its ambiguity.}
}

\newglossaryentry{entity}
{
  name={entity},
  description={A data structure as defined by a \gls{schema}.}
}

\newglossaryentry{entry}
{
  name={entry},
  plural={entries},
  description={An atomic \gls{entity}.}
}

\newglossaryentry{fieldset}
{
  name={fieldset},
  description={An atomic section of a UI form.}
}

\newglossaryentry{fieldset definition}
{
  name={fieldset definition},
  description={A structure which augments a \gls{section} \gls{schema} to define how to generate a \gls{fieldset}.}
}

\newglossaryentry{projection}
{
  name={projection},
  description={A \gls{derivation} which operates on a single \gls{section}, and thus can be run incrementally as \glspl{document} are added to the system.  However, since the \gls{derived dataset} is still non-canonical, it can be regenerated in its entirety from the \gls{canonical store} if necessary.}
}

\newglossaryentry{section}
{
  name={section},
  description={An \gls{entity} consisting of a collection of \glspl{entry}.  Broadly maps onto the set of data defined as an \gls{element} in the current implementation of OpenEyes.  The system defines subclasses of section for each specific use case.  A section type may be represented by one or more \glspl{fieldset definition} and \glspl{derivation}.}
}

\newglossaryentry{schema}
{
  name={schema},
  description={Describes the allowable format of a \gls{entity} at the database or API level.  Conformance to a schema means that the system will be capable of processing the \gls{entity}, but is not sufficient to guarantee that an \gls{entity} is valid as it may still fail to conform to local business rules.}
}
